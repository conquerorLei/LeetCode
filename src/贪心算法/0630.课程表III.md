#### 代码

```java
class Solution {
    public int scheduleCourse(int[][] courses) {
        Arrays.sort(courses, (a,b) -> a[1] - b[1]);
        
        PriorityQueue<Integer> pq = new PriorityQueue<>((a, b) -> b - a);

        int count = 0, ti, di;
        for(int[] course : courses){
            ti = course[0];di = course[1];
            if(count + ti <= di){
                count += ti;
                pq.offer(ti);
            }else if(!pq.isEmpty() && pq.peek() > ti){
                count -= pq.poll() - ti;
                pq.offer(ti);
            }
        }
        return pq.size();
    }
}
```
